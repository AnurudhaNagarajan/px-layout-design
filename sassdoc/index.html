    px-layout-design

    variables

    inuit-layout-namespace

    $inuit-layout-namespace: $inuit-namespace !default;

    Type

    String [default] - prepend value for all generated classes

    inuit-layout-min-width

    $inuit-layout-min-width: 300px !default;

    Type

    Number [default] - minimum width of individual layout item

    inuit-enable-layout--tiny

    $inuit-enable-layout--tiny: false !default;

    Type

    Boolean [default] - produce css for layout with tiny gutters between layout items?

    inuit-enable-layout--small

    $inuit-enable-layout--small: false !default;

    Type

    Boolean [default] - produce css for layout with small gutters between layout items?

    inuit-enable-layout--large

    $inuit-enable-layout--large: false !default;

    Type

    Boolean [default] - produce css for layout with large gutters between layout items?

    inuit-enable-layout--huge

    $inuit-enable-layout--huge: false !default;

    Type

    Boolean [default] - produce css for layout with huge gutters between layout items?

    inuit-enable-layout--flush

    $inuit-enable-layout--flush: false !default;

    Type

    Boolean [default] - produce css for layout with no gutters between layout items?

    inuit-enable-layout--rev

    $inuit-enable-layout--rev: false !default;

    Type

    Boolean [default] - produce css for layout items reversed?

    inuit-enable-layout--middle

    $inuit-enable-layout--middle: false !default;

    Type

    Boolean [default] - produce css to align layout items to the vertical centers of each other?

    inuit-enable-layout--bottom

    $inuit-enable-layout--bottom: false !default;

    Type

    Boolean [default] - produce css to align layout items to the vertical bottoms of each other?

    inuit-enable-layout--full

    $inuit-enable-layout--full: false !default;

    Type

    Boolean [default] - produce css to align layout items to all be the same height?

    inuit-enable-layout--right

    $inuit-enable-layout--right: false !default;

    Type

    Boolean [default] - produce css to make layout items fill up from the right hand side?

    inuit-enable-layout--center

    $inuit-enable-layout--center: false !default;

    Type

    Boolean [default] - produce css to make the layout items fill up from the center outward?

    inuit-enable-layout__item--full

    $inuit-enable-layout__item--full: false !default;

    Type

    Boolean [default] - produce css to make an individual layout item fill up the height of its layout?

    inuit-enable-layout__item--center

    $inuit-enable-layout__item--center: false !default;

    Type

    Boolean [default] - produce css to make an individual layout item fill up from the center vertically?

    inuit-enable-layout__item--bottom

    $inuit-enable-layout__item--bottom: false !default;

    Type

    Boolean [default] - produce css to make an individual layout item fill up from the bottom up?

    inuit-global-border-box

    $inuit-global-border-box: false !default;

    Type

    Boolean [default] - set a variable assuming that `box-sizing : border-box;` is not set globally. if it has been previously been defined, the following variable will be overriden and will be set to `true`.
